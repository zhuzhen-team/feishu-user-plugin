```javascript
// scripts/probe-feishu-docx.js
//
// One-shot: pull a representative document, run feishu-docx::transform, and
// emit a coverage report — what blocks appeared, what got dropped silently.
'use strict';

const fs = require('fs');
const path = require('path');
const { LarkOfficialClient } = require('../src/clients/official');
const credentials = require('../src/auth/credentials');
let docx;
try { docx = require('feishu-docx'); } catch (e) {
  console.error('feishu-docx not installed: npm install feishu-docx@^0.7.0'); process.exit(1);
}

(async () => {
  const docId = process.argv[2];
  if (!docId) { console.error('Usage: node scripts/probe-feishu-docx.js <docx_token | URL>'); process.exit(2); }

  const env = credentials.getActiveProfileEnv();
  const c = new LarkOfficialClient(env.LARK_APP_ID, env.LARK_APP_SECRET);
  if (env.LARK_USER_ACCESS_TOKEN) c.loadUAT();

  const blocks = await c.getDocBlocks(docId);
  const blockTypes = {};
  for (const b of blocks) blockTypes[b.block_type] = (blockTypes[b.block_type] || 0) + 1;
  console.log('Block types in document:', blockTypes);

  // Save fixture for later regression
  const fixtureDir = path.join(__dirname, '..', 'src', 'test-fixtures', 'doc-blocks');
  fs.mkdirSync(fixtureDir, { recursive: true });
  const fixturePath = path.join(fixtureDir, 'sample-1.json');
  fs.writeFileSync(fixturePath, JSON.stringify(blocks, null, 2));
  console.log(`Fixture saved: ${fixturePath}`);

  // Try transform
  const md = (typeof docx.transform === 'function') ? docx.transform(blocks) : (typeof docx.markdown === 'function' ? docx.markdown(blocks) : null);
  if (md === null) {
    console.error('Could not find transform function in feishu-docx; inspect API:');
    console.error(Object.keys(docx));
    process.exit(3);
  }

  const jsonSize = JSON.stringify(blocks).length;
  console.log(`JSON size:      ${jsonSize}`);
  console.log(`Markdown size:  ${md.length}`);
  console.log(`Ratio:          ${(md.length / jsonSize * 100).toFixed(1)}%  (target 30-50%)`);

  console.log('\n--- Markdown excerpt (first 500 chars) ---');
  console.log(md.slice(0, 500));
})().catch(e => { console.error(e); process.exit(1); });
```

```javascript
async read_doc_markdown(args, ctx) {
  const docId = await ctx.resolveDocId(args.document_id);
  const blocks = await ctx.getOfficialClient().getDocBlocks(docId);

  // Lazy-load feishu-docx (so test environments without the dep don't crash)
  let docx;
  try { docx = require('feishu-docx'); }
  catch (e) {
    return text(`read_doc_markdown: feishu-docx package not installed. Run: npm install feishu-docx@^0.7.0`);
  }

  const transformFn = typeof docx.transform === 'function' ? docx.transform
                    : typeof docx.markdown === 'function' ? docx.markdown
                    : null;
  if (!transformFn) {
    return text(`read_doc_markdown: feishu-docx version ${docx.version || 'unknown'} doesn't expose transform()/markdown(). Update or pin to ^0.7.0.`);
  }

  let md;
  try { md = transformFn(blocks); }
  catch (e) { return text(`read_doc_markdown: feishu-docx threw: ${e.message}`); }

  // Post-processor: normalise embedded image / file references.
  md = _normaliseEmbeds(md);

  return text(md);
},
```

Below the handlers, helper:

```javascript
function _normaliseEmbeds(md) {
  // 1. Image: feishu-docx may emit /image/<token> or similar. Convert to feishu://image_token/<token>.
  // 2. File: convert to feishu://file_token/<token>.
  // 3. Embedded sub-doc / bitable / sheet: feishu://base|sheet|docx|wiki/<token>.
  // (The exact replacements depend on what feishu-docx emits — adjust based on probe output.)
  return md
    .replace(/!\[([^\]]*)\]\(\/?image\/([a-zA-Z0-9_]+)\)/g, '![$1](feishu://image_token/$2)')
    // Add more substitutions if needed based on probe (Task 1)
    ;
}
```

```javascript
// src/test-read-doc-markdown.js
// Validates the markdown transformation against a saved fixture.
'use strict';

const fs = require('fs');
const path = require('path');
const assert = require('node:assert/strict');

function run() {
  const fixturePath = path.join(__dirname, 'test-fixtures', 'doc-blocks', 'sample-1.json');
  if (!fs.existsSync(fixturePath)) { console.log('read-doc-markdown: no fixture, skipping'); return; }
  const blocks = JSON.parse(fs.readFileSync(fixturePath, 'utf8'));
  let docx;
  try { docx = require('feishu-docx'); } catch (_) { console.log('read-doc-markdown: feishu-docx not installed, skipping'); return; }
  const transform = docx.transform || docx.markdown;
  if (!transform) { console.log('read-doc-markdown: no transform fn, skipping'); return; }

  const md = transform(blocks);
  assert.ok(typeof md === 'string', 'output should be a string');
  assert.ok(md.length > 0, 'output should be non-empty');

  // Token saving check: markdown should be smaller than JSON
  const jsonSize = JSON.stringify(blocks).length;
  const ratio = md.length / jsonSize;
  if (ratio > 0.6) {
    console.warn(`read-doc-markdown: ratio ${(ratio*100).toFixed(1)}% > 60% — consider tightening post-processor`);
  }
  console.log(`read-doc-markdown: PASS (ratio ${(ratio*100).toFixed(1)}%)`);
}

if (require.main === module) run();
module.exports = { run };
```

```javascript
// src/test-switch-profile.js
// e2e test: validate switch_profile invalidates cached clients + persists active.
//
// CAUTION: this test temporarily modifies ~/.feishu-user-plugin/credentials.json.
// Backup is taken at start and restored at end (try/finally). If the test
// crashes mid-run, the backup file is named cred-backup-<ts>.json — restore
// manually if you see one in the home dir.

'use strict';
const fs = require('fs');
const path = require('path');
const os = require('os');
const assert = require('node:assert/strict');

const credPath = path.join(os.homedir(), '.feishu-user-plugin', 'credentials.json');

function _readJson(p) {
  try { return JSON.parse(fs.readFileSync(p, 'utf8')); } catch (_) { return null; }
}

function _writeJson(p, obj) {
  fs.mkdirSync(path.dirname(p), { recursive: true, mode: 0o700 });
  fs.writeFileSync(p, JSON.stringify(obj, null, 2) + '\n', { mode: 0o600 });
}

async function run() {
  // STDERR warning to user
  console.error('[test-switch-profile] WARNING: temporarily modifying ~/.feishu-user-plugin/credentials.json. Backup at /tmp/cred-backup-<ts>.json. Stop running MCP processes (pkill -f feishu-user-plugin) to avoid interference.');

  const ts = Date.now();
  const backupPath = path.join(os.tmpdir(), `cred-backup-${ts}.json`);
  const originalCanonical = _readJson(credPath);

  // Backup
  if (originalCanonical) fs.copyFileSync(credPath, backupPath);
  else _writeJson(backupPath, { absent: true });
  console.error(`[test-switch-profile] backup → ${backupPath}`);

  try {
    // Ensure clean canonical with a default + an alt profile.
    const dummyAlt = {
      LARK_APP_ID: 'cli_test_alt_xxxxxxxx',
      LARK_APP_SECRET: 'test_secret_alt',
      LARK_COOKIE: 'session=fake_alt_cookie',
      LARK_USER_ACCESS_TOKEN: 'u-fake-alt-uat',
      LARK_USER_REFRESH_TOKEN: 'fake-alt-refresh',
    };
    const baseDefault = (originalCanonical?.profiles?.default) || {
      LARK_APP_ID: 'cli_test_def_xxxxxxxx',
      LARK_APP_SECRET: 'test_secret_def',
    };
    const fixture = {
      version: 1,
      active: 'default',
      profiles: { default: baseDefault, alt: dummyAlt },
      profileHints: {},
    };
    _writeJson(credPath, fixture);

    // Bust require cache so credentials.js re-reads fresh canonical
    delete require.cache[require.resolve('./auth/credentials')];
    delete require.cache[require.resolve('./server')];
    delete require.cache[require.resolve('./resolver')];
    delete require.cache[require.resolve('./tools/profile')];

    // Build ctx the same way server.js does, but stripped of the MCP transport.
    // Easiest path: require server.js and use its buildCtx helper if exported.
    // If not exported, replicate the minimal ctx surface.
    const credentials = require('./auth/credentials');
    const { LarkUserClient } = require('./clients/user');
    const { LarkOfficialClient } = require('./clients/official');
    let userClient = null;
    let officialClient = null;
    let currentProfile = credentials.getActiveProfileName();

    function profileEnv(name) { return credentials.getActiveProfileEnv(name); }
    async function getUserClient() {
      if (userClient) return userClient;
      const env = profileEnv(currentProfile);
      userClient = new LarkUserClient(env.LARK_COOKIE || 'dummy');
      // Don't await init() — that would hit the network with our dummy cookie.
      return userClient;
    }
    function getOfficialClient() {
      if (officialClient) return officialClient;
      const env = profileEnv(currentProfile);
      officialClient = new LarkOfficialClient(env.LARK_APP_ID, env.LARK_APP_SECRET);
      return officialClient;
    }
    const ctx = {
      getUserClient, getOfficialClient,
      listProfiles: () => credentials.listProfileNames(),
      getActiveProfile: () => currentProfile,
      setActiveProfile: (n) => {
        credentials.getActiveProfileEnv(n);
        currentProfile = n;
        userClient = null;
        officialClient = null;
        try { credentials.setActiveProfile(n); } catch (_) {}
      },
    };

    const profile = require('./tools/profile');

    // 1. Initial state
    assert.equal(currentProfile, 'default');
    const c1 = getOfficialClient();
    assert.ok(c1._appId.startsWith('cli_test_def_'), 'first client uses default APP_ID');

    // 2. switch_profile alt
    await profile.handlers.switch_profile({ name: 'alt' }, ctx);
    const fresh = _readJson(credPath);
    assert.equal(fresh.active, 'alt', 'credentials.json::active updated');

    // 3. Cached client invalidated; next get rebuilds
    const c2 = getOfficialClient();
    assert.notEqual(c2, c1, 'client instance should differ after switch');
    assert.ok(c2._appId.startsWith('cli_test_alt_'), 'rebuilt client uses alt APP_ID');

    // 4. switch back
    await profile.handlers.switch_profile({ name: 'default' }, ctx);
    assert.equal(_readJson(credPath).active, 'default');
    const c3 = getOfficialClient();
    assert.ok(c3._appId.startsWith('cli_test_def_'), 'switched back to default APP_ID');

    console.log('switch-profile-e2e: PASS');
  } finally {
    // Restore
    if (originalCanonical) {
      _writeJson(credPath, originalCanonical);
    } else {
      try { fs.unlinkSync(credPath); } catch (_) {}
    }
    console.error(`[test-switch-profile] restored credentials.json from ${backupPath}`);
  }
}

if (require.main === module) {
  run().catch((e) => {
    console.error('switch-profile-e2e: FAIL', e);
    process.exit(1);
  });
}
module.exports = { run };
```

```markdown
## switch_profile e2e (v1.3.9 F.1)

`src/test-switch-profile.js` validates that `switch_profile` correctly:
1. Atomically updates `~/.feishu-user-plugin/credentials.json::active`.
2. Invalidates the cached `userClient` / `officialClient` in the current process.
3. Rebuilds clients with the new profile's credentials on next access.

**The test temporarily modifies `~/.feishu-user-plugin/credentials.json`** — backup is automatic, restored on exit, but if the test crashes mid-run a `cred-backup-<ts>.json` will remain in `/tmp/`.

**Before running:** stop any running MCP processes (`pkill -f feishu-user-plugin`) to avoid them reacting to credential mutations via the v1.3.9 A.2 cross-process sync mechanism.

CI runs this test with no real Feishu credentials needed — uses dummy values that never hit the network.
```
