# v1.3.11 Lark Desktop Multi-Account Auto-Switch — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** When the user switches Lark Desktop to a different account, the MCP server automatically picks up that account's credentials — no CLI command, no MCP tool call required.

**Architecture:** Don't decrypt Lark's `cookie_store.db` (Keychain-keyed encryption, fragile). Instead, key off `mtime` of each account's `cookie_store.db`. Each MCP profile in `credentials.json` carries an optional `larkHash` field binding it to one of `~/Library/Containers/com.bytedance.macos.feishu/Data/Library/Application Support/LarkShell/sdk_storage/<hash>/`. The owner-process heartbeat (already 15 s tick) snapshots hash mtimes and switches the active profile when the most-recently-active hash differs. Cookies still come from each profile's own `LARK_COOKIE` (user provides them per profile, manual or via Playwright auto-capture).

**Tech Stack:** Node.js (no new deps — `fs.statSync` is enough), existing `src/events/owner.js` heartbeat, existing `src/auth/credentials.js` API surface.

**Platform scope:** macOS-only in v1.3.11; Linux/Windows return `null` from the platform detector and all reactor code no-ops.

```js
// src/auth/lark-desktop.js — Lark Desktop sdk_storage detection (v1.3.11).
//
// macOS-only in v1.3.11. Linux/Windows return null from getSdkStorageDir().
// We never read the encrypted cookie_store.db — only stat its mtime to detect
// account switches. Profile↔hash bindings live in credentials.json::profiles[*].larkHash.
'use strict';

const fs = require('fs');
const os = require('os');
const path = require('path');

const HASH_RE = /^[a-f0-9]{32}$/;

function _macSdkStorageDir() {
  return path.join(
    os.homedir(),
    'Library/Containers/com.bytedance.macos.feishu/Data/Library/Application Support/LarkShell/sdk_storage'
  );
}

function getSdkStorageDir() {
  if (process.platform !== 'darwin') return null;
  const dir = _macSdkStorageDir();
  try {
    return fs.statSync(dir).isDirectory() ? dir : null;
  } catch (_) {
    return null;
  }
}

function listAccountHashes({ dir } = {}) {
  const root = dir || getSdkStorageDir();
  if (!root) return [];
  let entries;
  try { entries = fs.readdirSync(root); } catch (_) { return []; }
  const out = [];
  for (const name of entries) {
    if (!HASH_RE.test(name)) continue;
    const accountDir = path.join(root, name);
    const dbPath = path.join(accountDir, 'cookie_store.db');
    let mtimeMs = 0;
    try { mtimeMs = fs.statSync(dbPath).mtimeMs; } catch (_) {
      // No cookie_store.db (account never logged in / cleared) — skip
      continue;
    }
    out.push({ hash: name, mtimeMs, dir: accountDir });
  }
  out.sort((a, b) => b.mtimeMs - a.mtimeMs);
  return out;
}

function mostRecentHash(opts = {}) {
  const list = listAccountHashes(opts);
  return list.length > 0 ? list[0] : null;
}

module.exports = {
  getSdkStorageDir,
  listAccountHashes,
  mostRecentHash,
  HASH_RE,
};
```

```js
// --- Lark Desktop hash bindings (v1.3.11 §A) ---
//
// Each profile may carry an optional larkHash 32-char-hex field binding it to
// one of ~/Library/Containers/com.bytedance.macos.feishu/.../sdk_storage/<hash>/.
// The owner heartbeat reactor uses this binding to decide which profile to
// switch to when the user changes account in Lark Desktop. See src/auth/lark-desktop.js.

const HASH_RE = /^[a-f0-9]{32}$/;

function getProfileLarkHash(name) {
  const f = _readFile();
  const target = name || (f ? f.active : 'default');
  if (f && f.profiles[target] && typeof f.profiles[target].larkHash === 'string') {
    return f.profiles[target].larkHash;
  }
  return null;
}

function setProfileLarkHash(name, hash) {
  if (hash !== null && (typeof hash !== 'string' || !HASH_RE.test(hash))) {
    throw new Error('setProfileLarkHash: hash must be 32-char hex (a-f, 0-9) or null');
  }
  const f = _readFile();
  if (!f) throw new Error('No credentials.json — cannot set profile larkHash.');
  if (!f.profiles[name]) {
    throw new Error(`setProfileLarkHash: profile "${name}" not found. Available: ${Object.keys(f.profiles).join(', ')}`);
  }
  if (hash === null) delete f.profiles[name].larkHash;
  else f.profiles[name].larkHash = hash;
  _atomicWriteJson(_credentialsPath(), f);
  return true;
}

function findProfileByHash(hash) {
  if (typeof hash !== 'string' || !HASH_RE.test(hash)) return null;
  const f = _readFile();
  if (!f) return null;
  for (const [name, profile] of Object.entries(f.profiles)) {
    if (profile.larkHash === hash) return name;
  }
  return null;
}
```

## Task 4: Heartbeat reactor in `src/server.js`

**Files:**
- Modify: `src/server.js`

The reactor has 3 responsibilities:
1. On every heartbeat, snapshot hash mtimes.
2. Detect when a non-active hash has the most-recent mtime AND its mtime advanced since prev snapshot → call `setActiveProfile(profile)` (existing credentials.json mtime path triggers `_maybeReconfigure`).
3. Detect freshly active hash that isn't bound to any profile → emit one-shot stderr hint per unknown hash per session.

- [ ] **Step 4.1 — Extract pure helpers above `_claimAndStart`**

Add right above `// --- Owner control loop (v1.3.9 A.1) ---` block:

```js
// --- Lark Desktop hash reactor state (v1.3.11) ---
let _lastHashMtimes = {};       // {hash: mtimeMs} from previous heartbeat
let _lastSwitchAt = 0;          // ms timestamp of last auto-switch (debounce)
const _seenUnboundHashes = new Set();  // print hint once per session per hash
const SWITCH_DEBOUNCE_MS = 5000;
const UNBOUND_FRESH_WINDOW_MS = 60_000;  // hash mtime within last 60s = "active"

function _snapshotHashMtimes() {
  const out = {};
  try {
    const ld = require('./auth/lark-desktop');
    for (const h of ld.listAccountHashes()) out[h.hash] = h.mtimeMs;
  } catch (_) {}
  return out;
}

// Returns { hash, profile } when a switch is warranted, else null.
// Side-effect: emits one-shot stderr hint when the most-recent hash is unbound.
function _detectLarkDesktopSwitch(prevSnapshot) {
  if (Date.now() - _lastSwitchAt < SWITCH_DEBOUNCE_MS) return null;
  const credsApi = require('./auth/credentials');
  const ld = require('./auth/lark-desktop');
  const list = ld.listAccountHashes();
  if (list.length === 0) return null;

  const top = list[0];
  const activeProfile = credsApi.getActiveProfileName();
  const activeHash = credsApi.getProfileLarkHash(activeProfile);
  if (top.hash === activeHash) return null;  // already on most-recent

  // The reactor only acts when top.mtime *advanced* since prev snapshot —
  // otherwise we'd repeatedly switch on a stable older snapshot.
  const prev = prevSnapshot[top.hash] || 0;
  if (top.mtimeMs <= prev) return null;

  const targetProfile = credsApi.findProfileByHash(top.hash);
  if (!targetProfile) {
    // Unbound active hash — print hint once per session.
    const isFresh = (Date.now() - top.mtimeMs) < UNBOUND_FRESH_WINDOW_MS;
    if (isFresh && !_seenUnboundHashes.has(top.hash)) {
      _seenUnboundHashes.add(top.hash);
      console.error(`[feishu-user-plugin] Lark Desktop active account hash ${top.hash} is not bound to any MCP profile. ` +
        `Run: npx feishu-user-plugin setup --profile <name> --bind-hash ${top.hash}`);
    }
    return null;
  }
  return { hash: top.hash, profile: targetProfile };
}
```

```js
// src/auth/lark-desktop.js — append:
const SWITCH_DEBOUNCE_MS = 5000;
const UNBOUND_FRESH_WINDOW_MS = 60_000;

// Pure-ish: takes injected `now`/`credsApi` for testability.
function detectSwitch({
  prevSnapshot,
  lastSwitchAt,
  seenUnboundHashes,
  credsApi = require('./credentials'),
  now = Date.now(),
  listFn = listAccountHashes,
  log = console.error,
} = {}) {
  if (now - lastSwitchAt < SWITCH_DEBOUNCE_MS) return { switchTo: null, isUnbound: false };
  const list = listFn();
  if (list.length === 0) return { switchTo: null, isUnbound: false };

  const top = list[0];
  const activeProfile = credsApi.getActiveProfileName();
  const activeHash = credsApi.getProfileLarkHash(activeProfile);
  if (top.hash === activeHash) return { switchTo: null, isUnbound: false };

  const prev = prevSnapshot[top.hash] || 0;
  if (top.mtimeMs <= prev) return { switchTo: null, isUnbound: false };

  const targetProfile = credsApi.findProfileByHash(top.hash);
  if (!targetProfile) {
    const isFresh = (now - top.mtimeMs) < UNBOUND_FRESH_WINDOW_MS;
    if (isFresh && seenUnboundHashes && !seenUnboundHashes.has(top.hash)) {
      seenUnboundHashes.add(top.hash);
      log(`[feishu-user-plugin] Lark Desktop active account hash ${top.hash} is not bound to any MCP profile. Run: npx feishu-user-plugin setup --profile <name> --bind-hash ${top.hash}`);
    }
    return { switchTo: null, isUnbound: true, hash: top.hash };
  }
  return { switchTo: { hash: top.hash, profile: targetProfile }, isUnbound: false };
}

module.exports = {

  SWITCH_DEBOUNCE_MS,
  UNBOUND_FRESH_WINDOW_MS,
};
```
