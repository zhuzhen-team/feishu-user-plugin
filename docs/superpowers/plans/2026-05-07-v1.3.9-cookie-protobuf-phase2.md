```javascript
// src/clients/image-meta.js — image metadata cache for cookie-gateway IMAGE sends.
//
// Schema: ~/.feishu-user-plugin/image-meta.json
//   { version: 1, entries: { <image_key>: { width, height, mime, size, uploaded_at } } }
//
// upload_image stores; send_image_as_user reads.

'use strict';
const fs = require('fs');
const path = require('path');
const os = require('os');
const imageSize = require('image-size').default || require('image-size');

const CAP = 1000;
const META_PATH = path.join(os.homedir(), '.feishu-user-plugin', 'image-meta.json');

function _readAll() {
  try {
    const raw = fs.readFileSync(META_PATH, 'utf8');
    const parsed = JSON.parse(raw);
    if (parsed?.version === 1 && parsed.entries && typeof parsed.entries === 'object') return parsed;
  } catch (_) {}
  return { version: 1, entries: {} };
}

function _writeAll(data) {
  const dir = path.dirname(META_PATH);
  try { fs.mkdirSync(dir, { recursive: true, mode: 0o700 }); } catch (_) {}
  // LRU prune
  const keys = Object.keys(data.entries);
  if (keys.length > CAP) {
    keys.sort((a, b) => (data.entries[a].uploaded_at || 0) - (data.entries[b].uploaded_at || 0));
    for (const k of keys.slice(0, keys.length - CAP)) delete data.entries[k];
  }
  const tmp = META_PATH + '.tmp.' + process.pid;
  fs.writeFileSync(tmp, JSON.stringify(data) + '\n', { mode: 0o600 });
  fs.renameSync(tmp, META_PATH);
}

function fromFile(filePath) {
  const stat = fs.statSync(filePath);
  if (stat.size > 50 * 1024 * 1024) throw new Error(`image too large for cache (${stat.size} bytes > 50 MB)`);
  const buf = fs.readFileSync(filePath);
  const dim = imageSize(buf);
  if (!dim || !dim.width || !dim.height) throw new Error(`could not parse image header from ${filePath}`);
  const mimeMap = { png: 'image/png', jpg: 'image/jpeg', jpeg: 'image/jpeg', gif: 'image/gif', webp: 'image/webp', bmp: 'image/bmp' };
  const mime = mimeMap[dim.type] || 'application/octet-stream';
  return { width: dim.width, height: dim.height, mime, size: stat.size };
}

function set(imageKey, meta) {
  if (!imageKey) return;
  const data = _readAll();
  data.entries[imageKey] = { ...meta, uploaded_at: Math.floor(Date.now() / 1000) };
  _writeAll(data);
}

function get(imageKey) {
  if (!imageKey) return null;
  const data = _readAll();
  return data.entries[imageKey] || null;
}

module.exports = { fromFile, set, get };
```

```javascript
async send_image_as_user(args, ctx) {
  const c = await ctx.getUserClient();
  const chatId = await _resolveCookieChatId(args.chat_id, ctx);
  const imageMeta = require('../clients/image-meta');
  const officialClient = ctx.getOfficialClient();

      imageKey = uploaded.image_key || uploaded;
      imageMeta.set(imageKey, meta);
    }
  }

  // Path B: explicit width/height/mime override (highest priority for non-path callers)
  if (args.width && args.height && args.mime) {
    meta = { width: args.width, height: args.height, mime: args.mime, size: args.size };
  } else if ((args.width || args.height || args.mime) && !meta) {
    return text('send_image_as_user: width/height/mime are all-or-nothing; pass all three or none.');
  }

  // Path C: cache lookup
  if (!meta && imageKey) meta = imageMeta.get(imageKey);

  if (!imageKey) {
    return text('send_image_as_user: image_key or image_path is required.');
  }
  if (!meta) {
    return text(

      'Pass one of:\n' +

      '  - width, height, mime          (explicit override; size optional)\n' +
      '  - re-upload via upload_image which now caches metadata\n' +
      'or fall back to send_message_as_bot(chat_id, msg_type="image", payload={image_key:"..."}).'
    );
  }

  const r = await c.sendImage(chatId, imageKey, {
    rootId: args.root_id,
    width: meta.width, height: meta.height,
    mime: meta.mime, size: meta.size,
```

```javascript
// src/test-protobuf-roundtrip.js
const fs = require('fs');
const path = require('path');
const protobuf = require('protobufjs');
const assert = require('node:assert/strict');

async function run() {
  const proto = await protobuf.load(path.join(__dirname, '..', 'proto', 'lark.proto'));
  const Packet = proto.lookupType('Packet');

  const fixtureDir = path.join(__dirname, '..', 'tests', 'fixtures', 'captured');
  let fixtures;
  try { fixtures = fs.readdirSync(fixtureDir); }
  catch (_) { console.log('protobuf-roundtrip: no fixtures, skipping'); return; }

  for (const f of fixtures.filter(n => n.endsWith('.b64'))) {
    const buf = Buffer.from(fs.readFileSync(path.join(fixtureDir, f), 'utf8').trim(), 'base64');
    const pkt = Packet.decode(buf);
    const re = Packet.encode(pkt).finish();
    // Soft assertion: decoded objects equal (byte-equality not guaranteed but field set should match)
    const reDecoded = Packet.decode(re);
    assert.deepEqual(
      Packet.toObject(pkt, { defaults: false }),
      Packet.toObject(reDecoded, { defaults: false }),
      `${f}: round-trip diverged`
    );
    console.log(`  ${f}: round-trip OK`);
  }
  console.log('protobuf-roundtrip: PASS');
}

if (require.main === module) run().catch(e => { console.error(e); process.exit(1); });
module.exports = { run };
```
