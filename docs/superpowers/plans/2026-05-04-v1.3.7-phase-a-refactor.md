- [ ] **Step 0.2: Write `scripts/smoke.js`**

This script spawns the MCP server, sends `tools/list`, captures the response, and emits a normalized JSON dump (sorted by name, with `inputSchema.required` sorted, etc.). It also calls `get_login_status` and dumps the keys (not values) of the response. The output is stable across runs as long as no tool is added/removed/renamed and no schema changes.

**Cred sourcing:** `src/index.js` only reads `process.env.LARK_*` and won't auto-discover `~/.claude.json`. So this script MUST inject creds itself via `readCredentials()` (from `src/config.js`) before spawning the child — otherwise the baseline captures "all NO creds" warnings and `get_login_status` returns the not-configured shape, which is stable but doesn't exercise the full code path.

```js
#!/usr/bin/env node
// scripts/smoke.js — MCP smoke test for refactor before/after diff.
// Spawns src/index.js as a child via stdio, sends:
//   1. tools/list                   → dumps sorted (name, description, inputSchema) to stdout
//   2. tools/call get_login_status  → dumps recursive Object.keys (not values) to stdout
// Exits 0 on success, 1 on protocol error. Diff output against tests/baseline/*.json.
//
// Usage:
//   node scripts/smoke.js dump            # print normalized current snapshot to stdout
//   node scripts/smoke.js diff            # compare against tests/baseline/* and exit non-zero on mismatch
//   node scripts/smoke.js write-baseline  # overwrite tests/baseline/*.json with current snapshot

const { spawn } = require('child_process');
const path = require('path');
const fs = require('fs');
const { readCredentials } = require('../src/config');

const SERVER_PATH = path.join(__dirname, '..', 'src', 'index.js');
const BASELINE_DIR = path.join(__dirname, '..', 'tests', 'baseline');

function jsonrpc(id, method, params) {
  return JSON.stringify({ jsonrpc: '2.0', id, method, params }) + '\n';
}

function normalizeSchema(s) {
  if (!s || typeof s !== 'object') return s;
  if (Array.isArray(s)) return s.map(normalizeSchema);
  const out = {};
  for (const k of Object.keys(s).sort()) {
    out[k] = k === 'required' && Array.isArray(s[k]) ? [...s[k]].sort() : normalizeSchema(s[k]);
  }
  return out;
}

function shapeOnly(v) {
  if (v === null || v === undefined) return typeof v;
  if (Array.isArray(v)) return v.length === 0 ? '[]' : ['<' + typeof v[0] + '>'];
  if (typeof v === 'object') {
    const out = {};
    for (const k of Object.keys(v).sort()) out[k] = shapeOnly(v[k]);
    return out;
  }
  return typeof v;
}

async function runSmoke() {
  // Inject creds from ~/.claude.json so the spawned MCP server has the same auth
  // it would have when launched by Claude Code. Without this, baseline captures
  // "Cookie=NO App=NO UAT=NO" warnings and login_status is the not-configured shape.
  const creds = readCredentials() || {};
  const childEnv = { ...process.env };
  for (const k of ['LARK_COOKIE', 'LARK_APP_ID', 'LARK_APP_SECRET', 'LARK_USER_ACCESS_TOKEN', 'LARK_USER_REFRESH_TOKEN', 'LARK_PROFILES_JSON']) {
    if (creds[k] && !childEnv[k]) childEnv[k] = creds[k];
  }
  const child = spawn('node', [SERVER_PATH], {
    stdio: ['pipe', 'pipe', 'pipe'],
    env: childEnv,
  });
  let buf = '';
  const responses = new Map();
  child.stdout.on('data', (d) => {
    buf += d.toString();
    const lines = buf.split('\n');
    buf = lines.pop();
    for (const line of lines) {
      if (!line.trim()) continue;
      try {
        const msg = JSON.parse(line);
        if (msg.id != null) responses.set(msg.id, msg);
      } catch {}
    }
  });
  child.stderr.on('data', () => {}); // discard

  child.stdin.write(jsonrpc(1, 'initialize', { protocolVersion: '2024-11-05', capabilities: {}, clientInfo: { name: 'smoke', version: '0' } }));
  await waitFor(() => responses.has(1), 5000);

  child.stdin.write(jsonrpc(2, 'tools/list', {}));
  await waitFor(() => responses.has(2), 5000);

  child.stdin.write(jsonrpc(3, 'tools/call', { name: 'get_login_status', arguments: {} }));
  await waitFor(() => responses.has(3), 10000);

  child.kill('SIGTERM');

  const tools = (responses.get(2)?.result?.tools || []).map((t) => ({
    name: t.name,
    description: t.description,
    inputSchema: normalizeSchema(t.inputSchema),
  })).sort((a, b) => a.name.localeCompare(b.name));

  let loginShape = null;
  try {
    const txt = responses.get(3)?.result?.content?.[0]?.text;
    loginShape = shapeOnly(JSON.parse(txt));
  } catch {
    loginShape = { _error: 'get_login_status returned non-json or failed' };
  }

  return { tools, loginShape };
}

function waitFor(fn, timeoutMs) {
  return new Promise((resolve, reject) => {
    const start = Date.now();
    const tick = () => {
      if (fn()) return resolve();
      if (Date.now() - start > timeoutMs) return reject(new Error('timeout'));
      setTimeout(tick, 50);
    };
    tick();
  });
}

(async () => {
  const cmd = process.argv[2] || 'dump';
  const snap = await runSmoke();
  const toolsPath = path.join(BASELINE_DIR, 'tools-list.json');
  const loginPath = path.join(BASELINE_DIR, 'login-status-shape.json');

  if (cmd === 'dump') {
    process.stdout.write(JSON.stringify(snap, null, 2) + '\n');
    return;
  }
  if (cmd === 'write-baseline') {
    fs.mkdirSync(BASELINE_DIR, { recursive: true });
    fs.writeFileSync(toolsPath, JSON.stringify(snap.tools, null, 2) + '\n');
    fs.writeFileSync(loginPath, JSON.stringify(snap.loginShape, null, 2) + '\n');
    console.error(`Baseline written: ${snap.tools.length} tools`);
    return;
  }
  if (cmd === 'diff') {
    const expectedTools = JSON.parse(fs.readFileSync(toolsPath, 'utf8'));
    const expectedLogin = JSON.parse(fs.readFileSync(loginPath, 'utf8'));
    const actualToolsStr = JSON.stringify(snap.tools, null, 2);
    const expectedToolsStr = JSON.stringify(expectedTools, null, 2);
    if (actualToolsStr !== expectedToolsStr) {
      console.error('TOOLS MISMATCH');
      console.error(`expected ${expectedTools.length} tools, got ${snap.tools.length}`);
      process.exit(1);
    }
    if (JSON.stringify(snap.loginShape) !== JSON.stringify(expectedLogin)) {
      console.error('LOGIN STATUS SHAPE MISMATCH');
      console.error('expected:', JSON.stringify(expectedLogin, null, 2));
      console.error('actual:  ', JSON.stringify(snap.loginShape, null, 2));
      process.exit(1);
    }
    console.error(`OK: ${snap.tools.length} tools, login_status shape matches`);
    return;
  }
  console.error('usage: smoke.js [dump|diff|write-baseline]');
  process.exit(2);
})();
```

```js
// from this process or its dependencies corrupts the transport. Redirect to stderr.

console.log = (...args) => console.error(...args);
console.info = (...args) => console.error(...args);

// Stderr-only logger for the Lark SDK (the SDK's defaultLogger.error() writes to
// stdout via console.log, which would also corrupt MCP stdio).
const stderrLogger = {
  debug: (...a) => console.error('[lark-sdk]', ...a),
  info: (...a) => console.error('[lark-sdk]', ...a),
  warn: (...a) => console.error('[lark-sdk]', ...a),
  error: (...a) => console.error('[lark-sdk]', ...a),
```

```js
// src/tools/_registry.js — shared infrastructure for tool modules.

// Response builders — every handler returns one of these.
const text = (s) => ({ content: [{ type: 'text', text: s }] });
```

```js
// src/server.js — MCP bootstrap, tool registration, request dispatch.
const { Server } = require('@modelcontextprotocol/sdk/server/index.js');
const { StdioServerTransport } = require('@modelcontextprotocol/sdk/server/stdio.js');
const { CallToolRequestSchema, ListToolsRequestSchema } = require('@modelcontextprotocol/sdk/types.js');
const path = require('path');
require('dotenv').config({ path: path.join(__dirname, '..', '.env') });

const { LarkUserClient } = require('./clients/user');
const { LarkOfficialClient } = require('./clients/official');
const { resolveToObj, resolveToken, parseFeishuInput } = require('./resolver');

// Lazy-load tool modules so missing optional deps don't crash startup.
const TOOL_MODULES = [
  require('./tools/profile'),
  require('./tools/diagnostics'),
  require('./tools/calendar'),
  require('./tools/okr'),
  require('./tools/wiki'),
  require('./tools/contacts'),
  require('./tools/groups'),
  require('./tools/uploads'),
  require('./tools/drive'),
  require('./tools/im-read'),
  require('./tools/docs'),
  require('./tools/messaging-bot'),
  require('./tools/messaging-user'),
  require('./tools/bitable'),
];

const SCHEMAS = TOOL_MODULES.flatMap((m) => m.schemas);
const HANDLERS = Object.fromEntries(TOOL_MODULES.flatMap((m) => Object.entries(m.handlers)));

// Cached client factory — moved from old index.js
const profileMap = (() => { try { return JSON.parse(process.env.LARK_PROFILES_JSON || '{}'); } catch { return {}; } })();
let activeProfile = 'default';
function profileEnv(name) {
  if (name === 'default' || !profileMap[name]) return process.env;
  return { ...process.env, ...profileMap[name] };
}
let _userClient, _officialClient;
async function getUserClient() {
  if (_userClient) return _userClient;
  const env = profileEnv(activeProfile);
  _userClient = new LarkUserClient(env.LARK_COOKIE);
  await _userClient.init?.();
  return _userClient;
}
function getOfficialClient() {
  if (_officialClient) return _officialClient;
  const env = profileEnv(activeProfile);
  _officialClient = new LarkOfficialClient({
    appId: env.LARK_APP_ID,
    appSecret: env.LARK_APP_SECRET,
    userAccessToken: env.LARK_USER_ACCESS_TOKEN,
    userRefreshToken: env.LARK_USER_REFRESH_TOKEN,
  });
  return _officialClient;
}
function switchProfile(name) {
  if (name !== 'default' && !profileMap[name]) throw new Error(`Unknown profile: ${name}`);
  activeProfile = name;
  _userClient = null;
  _officialClient = null;
}
async function resolveDocId(input) { return resolveToObj(input, getOfficialClient()); }

const server = new Server(
  { name: 'feishu-user-plugin', version: require('../package.json').version },
  { capabilities: { tools: {} } }
);

server.setRequestHandler(ListToolsRequestSchema, async () => ({ tools: SCHEMAS }));

server.setRequestHandler(CallToolRequestSchema, async (req) => {
  const { name, arguments: args } = req.params;
  const handler = HANDLERS[name];
  if (!handler) return { content: [{ type: 'text', text: `Unknown tool: ${name}` }], isError: true };
  const ctx = {
    getUserClient,
    getOfficialClient,
    switchProfile,
    profileMap,
    activeProfile: () => activeProfile,
    resolveDocId,
    resolveToken,
    parseFeishuInput,
  };
  try {
    return await handler(args || {}, ctx);
  } catch (err) {
    return { content: [{ type: 'text', text: `Error: ${err.message || err}` }], isError: true };
  }
});

async function main() {
  const transport = new StdioServerTransport();
  await server.connect(transport);
  console.error('[feishu-user-plugin] MCP server ready');
}

module.exports = { main, SCHEMAS };

if (require.main === module) {
  main().catch((err) => { console.error('Fatal:', err); process.exit(1); });
}
```

```markdown
# v1.3.7 Refactor Notes — Where New Code Goes

## Goals
- One file = one domain. Max 600 lines as a soft target; >900 lines is a smell.
- `index.js` stays a 5-line entry point. Never add logic there again.
- `server.js` stays a thin bootstrap + dispatcher. Never add tool-specific logic there.

## Decision Tree for "Where does my new code go?"

### New tool (handler + schema)
- Determine its **domain** (messaging? bitable? calendar? ...).
- Add schema to `src/tools/<domain>.js::schemas`.
- Add handler to `src/tools/<domain>.js::handlers`.
- If the domain doesn't exist yet, only create a new `tools/<x>.js` if you're adding ≥3 related tools. Otherwise piggyback on the closest existing one.

### New API call to Feishu Official API
- Add the method to `src/clients/official/<domain>.js`.
- If the call is shared across multiple domains, put it in `clients/official/base.js`.

### New API call via Cookie identity
- Add to `src/clients/user.js`.

### New auth / credential logic
- `src/auth/cookie.js` for cookie session lifecycle.
- `src/auth/uat.js` for UAT refresh / file lock / OAuth token rotation.
- `src/auth/credentials.js` for credential storage / single-source-of-truth (Phase B).

### New config / setup behaviour
- `src/config/discovery.js` for "where does the config live" questions.
- `src/config/persistence.js` for atomic write / merge logic.
- `src/config/setup.js` for the `setup` CLI command implementation.

### New cross-cutting helper
- If used by ≥2 modules: `src/utils.js`.
- If a tool-specific helper used by only one tool: keep inside that tool file.

## What NOT to Do
- Do NOT add new methods to `src/official.js` or `src/index.js` — they are barrel/entry shells.
- Do NOT create `src/tools/<x>.js` for a single tool. Group related tools.
- Do NOT bypass `src/server.js` to register tools. Every handler must be reachable via `TOOL_MODULES.flatMap(m => m.schemas)`.
- Do NOT bypass `src/clients/official/index.js` to construct the client. Always `require('./clients/official')`.

## When These Rules Don't Fit
If a feature genuinely doesn't fit any domain (e.g. WebSocket event subscription in v1.3.8), create a new top-level subdirectory (`src/events/`, `src/realtime/`) and add it to the decision tree above.
```
