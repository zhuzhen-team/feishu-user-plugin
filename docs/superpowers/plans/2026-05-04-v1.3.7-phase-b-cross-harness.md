- [ ] **B1.3: Write `src/prompts/_registry.js`.**
  ```js
  const fs = require('fs');
  const path = require('path');
  const SKILLS_DIR = path.join(__dirname, '..', '..', 'skills', 'feishu-user-plugin');

  function parseFrontmatter(content) {
    const m = content.match(/^---\n([\s\S]*?)\n---\n([\s\S]*)$/);
    if (!m) return { meta: {}, body: content };
    const meta = {};
    for (const line of m[1].split('\n')) {
      const kv = line.match(/^(\w+):\s*(.*)$/);
      if (kv) meta[kv[1]] = kv[2].trim();
    }
    return { meta, body: m[2].trim() };
  }

  function loadAllSkills() {
    const files = fs.readdirSync(SKILLS_DIR).filter(f => f.endsWith('.md') && f !== 'SKILL.md');
    return files.map(f => {
      const content = fs.readFileSync(path.join(SKILLS_DIR, f), 'utf8');
      const { meta, body } = parseFrontmatter(content);
      return {
        name: meta.name || path.basename(f, '.md'),
        description: meta.description || '',
        arguments: parseArguments(body),
        body,
      };
    });
  }

  function parseArguments(body) {
    // Scan body for {{var}} placeholders and return [{name, description, required}].
    // Simpler heuristic: any {{ident}} → required arg with the ident name.
    const matches = [...body.matchAll(/\{\{(\w+)\}\}/g)];
    const seen = new Set();
    const args = [];
    for (const m of matches) {
      if (seen.has(m[1])) continue;
      seen.add(m[1]);
      args.push({ name: m[1], description: '', required: true });
    }
    return args;
  }

  module.exports = { loadAllSkills };
  ```

- [ ] **B1.4: Write `src/prompts/index.js`.**
  ```js
  const { loadAllSkills } = require('./_registry');

  const PROMPTS = loadAllSkills();
  const PROMPT_BY_NAME = Object.fromEntries(PROMPTS.map(p => [p.name, p]));

  function listPrompts() {
    return PROMPTS.map(p => ({
      name: p.name,
      description: p.description,
      arguments: p.arguments,
    }));
  }

  function getPrompt(name, args = {}) {
    const p = PROMPT_BY_NAME[name];
    if (!p) throw new Error(`Unknown prompt: ${name}`);
    let text = p.body;
    for (const arg of p.arguments) {
      const val = args[arg.name] || `<${arg.name}>`;
      text = text.replaceAll(`{{${arg.name}}}`, val);
    }
    return {
      description: p.description,
      messages: [{ role: 'user', content: { type: 'text', text } }],
    };
  }

  module.exports = { listPrompts, getPrompt, PROMPTS };
  ```

## Sub-PR B2 — Single-Source Credentials File

**Why:** Today `findMcpConfig()` returns the FIRST config that matches (`~/.claude.json` top-level, then `projects[*]`, then `~/.codex/config.toml`, etc). When the user has feishu-user-plugin configured in both Claude Code and Codex, cookie heartbeat + UAT refresh only update the first match. The other harness gets stale tokens until the next OAuth re-run. This sub-PR fixes that by introducing a canonical credentials file that all harness configs point at.

### File Structure

```
~/.feishu-user-plugin/credentials.json    # 0600 perms — single source of truth
src/
├── auth/
│   ├── credentials.js                    # Phase A stub → real impl
│   ├── cookie.js                         # heartbeat (extracted from clients/user.js)
│   └── uat.js                            # UAT refresh + lock (extracted from clients/official/base.js)
├── config/
│   ├── discovery.js                      # findMcpConfig — used only for one-time migration
│   ├── persistence.js                    # atomic write to credentials.json
│   └── setup.js                          # writes pointer envs to harness configs
├── cli.js                                # add `migrate` subcommand
└── clients/
    ├── user.js                           # heartbeat → require('../auth/cookie')
    └── official/base.js                  # UAT refresh → require('../../auth/uat')
```

### credentials.json Schema

```json
{
  "version": 1,
  "active": "default",
  "profiles": {
    "default": {
      "LARK_COOKIE": "session=...",
      "LARK_APP_ID": "cli_xxx",
      "LARK_APP_SECRET": "yyy",
      "LARK_USER_ACCESS_TOKEN": "u-xxx",
      "LARK_USER_REFRESH_TOKEN": "r-xxx",
      "LARK_UAT_EXPIRES": 1735689600
    },
    "alt": { ... }
  },
  "profileHints": {
    "doctok_xxx": "alt",
    "oc_xxx": "default"
  }
}
```

`profileHints` is the cache for Phase v1.3.9's auto-switch (already documented in ROADMAP.md). Phase B2 only initializes the structure; Phase v1.3.9 fills it.

## Sub-PR B3 — Sync Hooks

**Why:** The project rule "every CLAUDE.md change must propagate to AGENTS.md + skills/feishu-user-plugin/references/CLAUDE.md + team-skills repo" is enforced manually today and gets forgotten. Likewise tool-count badge in README, version pins across `package.json` + `plugin.json` + `SKILL.md`. Move all of this into git hooks so the next contributor can't merge without sync.

### File Structure

```
.husky/
├── pre-commit            # runs sync scripts
├── commit-msg            # validates commit message format (optional)
└── post-merge            # syncs team-skills repo on main updates (optional)
scripts/
├── sync-claude-md.sh     # CLAUDE.md → AGENTS.md + skill reference
├── check-version.js      # verify package.json + plugin.json + SKILL.md match
├── check-tool-count.js   # verify README badge matches src/server.js TOOLS.length
└── sync-team-skills.sh   # cp -r skills/ to team-skills repo, push branch + auto-merge
package.json              # add husky devDep + prepare script
```

### Tasks

- [ ] **B3.1: Install husky.**
  ```bash

  npm install --save-dev husky
  npx husky init
  ```
  This creates `.husky/pre-commit` and adds `prepare` script to package.json.

- [ ] **B3.2: Write `scripts/sync-claude-md.sh`.**
  ```bash
  #!/usr/bin/env bash
  set -e
  ROOT="$(git rev-parse --show-toplevel)"
  cd "$ROOT"
  if git diff --cached --name-only | grep -qx "CLAUDE.md"; then
    tail -n +2 CLAUDE.md > /tmp/feishu-claude-body.md

    git add AGENTS.md skills/feishu-user-plugin/references/CLAUDE.md
    echo "[hook] CLAUDE.md → AGENTS.md + skill reference synced"
  ```

  ```js
  const fs = require('fs');
  ```
